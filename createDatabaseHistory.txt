CREATE SCHEMA sql_tutoring;

CREATE TABLE sql_tutoring.`users` (
  idUser INT NOT NULL,
  email VARCHAR(45) NOT NULL,
  name VARCHAR(45) NOT NULL,
  password VARCHAR(64) NOT NULL,
  isAdmin TINYINT NULL DEFAULT 0,
  PRIMARY KEY (idUser, email),
  UNIQUE INDEX idUser_UNIQUE (idUser ASC) VISIBLE,
  UNIQUE INDEX email_UNIQUE (email ASC) VISIBLE);

CREATE TABLE sql_tutoring.`quiz_questions`  (
    id INT NOT NULL PRIMARY KEY,
    title VARCHAR(255),
    content TEXT,
    category VARCHAR(255),
    difficulty VARCHAR(50),
    hint TEXT
);

CREATE TABLE sql_tutoring.`quiz_answers` (
    idAnswer INT AUTO_INCREMENT PRIMARY KEY,
    question_id INT,
    answer VARCHAR(255),
    FOREIGN KEY (question_id) REFERENCES quiz_questions(id)
);
